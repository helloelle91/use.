    sum=0
    for page2 in G:
        if page in G[page2]:
            sum += pr[page2]/len(G[page2])
    new_pr[page] = (1-d)+d*sum

for page in pr:

def leven(a,b):
    n = len(a)
    m = len(b)
    
    A=[[i+j for j in range(m+1)]for i in range(n+1)]
    
    for i in range(n):
        for j in range(m):
            A[i+1][j+1]=min(A[i][j+1]+1,
                           A[i+1][j]+1,
                           A[i][j]+(int(a[i]!=b[j])))
    return A[n][m]

def Js(doc1,doc2):
    A = set(doc1.lower().split())
    B = set(doc2.lower().split())
    
    intersection = A.intersection(B)
    union = A.union(B)
    
    return (len(intersection)/len(union))

from functools import reduce
from collections import defaultdict

def mapper(data):
    char_count = defaultdict(int)
    for char in data:
        char_count[char.lower()] += 1
    return (char_count.items())

def reducer(count1,count2):
    merged_result = defaultdict(int)
    
    for char,count in count1:
        merged_result[char] += count
        
    for char,count in count2:
        merged_result[char] += count
    return (merged_result.items()) 


data = "You can always add your own word vectors, or use one of the larger models"
chunks = [chunk for chunk in data]

mapped_result = map(mapper, chunks)

reducer_result = reduce(reducer, mapped_result)

for char,count in reducer_result:
    print(f"Characters : {char} Count : {count}")

import networkx as nx 
import xml.etree.ElementTree as et

def parse(xml_data):
    root = et.fromstring(xml_data)
    return root

def generate(xml_root):
    G = nx.DiGraph()
    
    for page in xml_root.findall('.//page'):
        node = page.find('id').text
        G.add_node(node)
        
        links = page.findall('.//link')
        for link in links:
            edge = link.text
            G.add_edge(node, edge)
    return G
    
def pr(graph, topic):
    alpha = 0.85
    per = {node: 1.0 if node in topic else 0.0 for node in graph.nodes}
    max_iter = 100
    tol = 1e-6
    return nx.pagerank(graph, alpha, per, max_iter, tol)
